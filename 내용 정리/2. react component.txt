import React, {Component} from 'react';

class Subject extends Component{
	render(){
		return(
			//하나 이상의 태그를 포함한 html
			<header>
				<h1>WEB</h1>
				world wide web!
			<header>
		);
	}
}

#App class에 추가

class App extends Component{
	render(){
		return(
			<div className="App">
				<Subject></Subject>
			</div>
		);
	}
}

#사용자 정의 컴포넌트 만들기

class Subject extends Component{
	render(){
		return(
			<header>
				<h1>{this.props.title}</h1>
				{this.props.contents}
			</header>
		);
	}
}

class App extends Component{
	render(){
		return(
			<Subject title="WEB" contents="world wide web!"></Subject>
		);
	}
}

15. state (with props)

#react의 여러 기능 조작 장치 같은 개념

#props를 좀 더 효율적 사용하기

class App extends Components{
	constructor(props){
		super(props);
		this.state={
			subject:{title:'WEB', sub:'World Wide Web!'}
		}
	}

	render(){
		return(
			<div className='App'>
				<Subject>
					title={this.state.subject.title} //위의 생성자에서 생성한 state 사용
					contents={this.state.subject.sub}
				</Subject>
			</div>
		);
	}
}

export default App;
